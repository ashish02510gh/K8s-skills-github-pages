3. Bare-Metal Kubernetes Installation

ðŸ‘‰ Click to Expand

To install Kubernetes on bare metal, we discussed the following steps:

  1. Prepare the host machines with an operating system like Ubuntu or CentOS.
  2. Install Docker as the container runtime.
  3. Install Kubernetes components (kubeadm, kubectl, kubelet).
  4. Initialize the cluster using kubeadm:
    5. sudo kubeadm init --pod-network-cidr=192.168.0.0/16
  5. Set up networking with a CNI plugin like Calico:
    6. kubectl apply -f https://docs.projectcalico.org/v3.14/manifests/calico.yaml

- name: Install Kubernetes
  hosts: all
  tasks:
    - name: Install Docker
      apt:
        name: docker.io
        state: present
    
    - name: Install Kubernetes packages
      apt:
        name:
          - kubelet

This playbook installs Docker and Kubernetes, then initializes the cluster.

5. Terraform Example: Azure VNet with NSG, Subnets, and VPN Gateways

ðŸ‘‰ Click to Expand

The Terraform example for setting up an Azure Virtual Network includes Resource Groups, Subnets, NSGs, and HTTP traffic rules: 

provider "azurerm" {
  features = {}
}

resource "azurerm_resource_group" "example" {
  name     = "example-resource-group"
  location = "East US"
}

resource "azurerm_virtual_network" "example_vnet" {
  name                = "example-vnet"
  location            = azurerm_resource_group.example.location
  resource_group_name = azurerm_resource_group.example.name
  address_space       = ["10.0.0.0/16"]

  tags = {
    Environment = "Dev"
  }
}

resource "azurerm_subnet" "example_subnet" {
  name                 = "example-subnet"
  resource_group_name  = azurerm_resource_group.example.name
  virtual_network_name = azurerm_virtual_network.example_vnet.name
  address_prefixes     = ["10.0.1.0/24"]
}

resource "azurerm_network_security_group" "example_nsg" {
  name                = "example-nsg"
  location            = azurerm_resource_group.example.location
  resource_group_name = azurerm_resource_group.example.name

  security_rule {
    name                       = "Allow_HTTP"
    priority                   = 100
    direction                  = "Inbound"
    access                     = "Allow"
    protocol                   = "Tcp"
    source_port_range          = "*"
    destination_port_range     = "80"
    source_address_prefix      = "*"
    destination_address_prefix = "*"
  }
}
